Default parameters define value if value is not given while calling the function (or undefined passed).

  • Evaulated in runtime (when function is called)
  • It can be based on other default parameters (firstLetter based on word in example)
  • Explicit undefined defers to default
var extractFirst = function(word) {
	return word.substring(0, 1);
},
isFirstCapital = function(word, firstLetter = extractFirst(word)) {
	if (firstLetter.uppercase() === firstLetter) {
		return true;
	}
};

AppLogger.info(isFirstCapital('Slaven'));
AppLogger.info(isFirstCapital('slaven'));
AppLogger.info(isFirstCapital('slaven', 'S'));
