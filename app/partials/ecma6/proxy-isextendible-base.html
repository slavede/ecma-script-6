• Trap for Object.isExtensible()
  • Object.isExtensible() determines if an object is extensible. By default all objects are. There are few ways of making it non extensible (Object.preventExtension(), Object.seal(), Object.freeze().
  • Proxy must return same extensibility as target object
AppLogger.info('Object.isExtensible(myObject): ' + Object.isExtensible(myObject));

var extensibleProxy = new Proxy(myObject, {
	isExtensible : function(target) {
		AppLogger.info('I am being proxied through extensibleProxy' + target.name);
		return true;
	}
});

AppLogger.info('Object.isExtensible(myObject): ' + Object.isExtensible(myObject));
AppLogger.info('Object.isExtensible(extensibleProxy): ' + Object.isExtensible(extensibleProxy));
