String literals that allow string interpolation and multiline strings.

  • Template strings are enclosed with a back-tick (`) with placeholders enclosed with ${variable}
  • They can be multilined without any special sign for new line
  • Tagged functions allow you to pass values to function instead of default concatenation
    • First parameter is array of string from original enclosed string
    • First parameter has also raw property (not interpolated)
    • Rest of params are variables from placeholders (enclosed by ${}). Useful to use rest parameters for them (...). See Rest Parameter Examples for more details
var name = "Slaven", surname = "Tomac";

AppLogger.info(`Hello there ${name} ${surname} !`);


var multilineString = `This is my multiline string ${name}
						and what will you do about it?

						Ha?!
						`;

AppLogger.info(multilineString);

var a = 1;
AppLogger.info(`I can also calculate in string now : ${5 + a}`);

function myTagger(strings, value1, value2) {
	return 'raw: ' + strings.raw.join(',') + '\n strings: ' + strings.join(',') + '\n values: ' + value1 + ' - ' + value2 ;
}

AppLogger.info(myTagger`User ${name} will use\n my tagger here for ${a}st time`);
