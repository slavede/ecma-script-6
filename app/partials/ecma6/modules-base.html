ES5 doesn't have built in support for modules which is solved by using CommonJS Modules (e.g. NodeJS modules) and AMD modules (e.g. RequireJS). ES6 gets this built in

  • With named exports we can export/import more variables (mylogger.js in examples)
  • Default exports defines what's being imported by default into other module (sum.js). Only one per module
  • Combination of both is possible as well

mylogger.js

export function loggerInterceptor(logger, message) {
	logger.info('loggerInterceptor --- ' + message);
};

export var variablePi = '3.141593';

sum.js

export default function (a, b, c) { 
	return a + b + c;
};

importer.js

import {variablePi,loggerInterceptor} from 'mylogger';
import sumCalculation from 'sum';

AppLogger.info(variablePi);
loggerInterceptor(AppLogger, 'importer call')
AppLogger.info(sumCalculation(1,2,3));
