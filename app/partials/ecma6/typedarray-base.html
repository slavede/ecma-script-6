Allows declaring array with specicic type, length and/or buffer.

  • Few constructors:
    • new TypedArray(length)
    • new TypedArray(typedArray)
    • new TypedArray(object)
    • new TypedArray(buffer [,bytesOffset [,length]])
    where
    • length - if given, array with that many zeroes will be created
    • typedArray - it can be object of any of the typed arrays, copies (and converts) values into new array
    • object - creates array from the object
    • TypedArray can be one of the following:
      • Int8Array()
      • Uint8Array()
      • Uint8ClampedArray() - values clamped to 0-255
      • Int16Array()
      • Uint16Array()
      • Int32Array()
      • Uint32Array()
      • Float32Array()
      • Float64Array()
  • They don't support .push()!!! :o

Adding/converting elements

  • Initially it's an array with zeroes 
    var int8Array = new Int8Array(8);
AppLogger.info(int8Array.join(','));
  • To add one element you can't use push as in normal array types, but assign it via index access 
    int8Array[0] = 1;
AppLogger.info(int8Array.join(','));
int8Array[1] = -123;
AppLogger.info(int8Array.join(','));
int8Array[1] = -128;
AppLogger.info(int8Array.join(','));
  • If assigned value out of the scope it will be modulo modified 
    int8Array[1] = -129;
AppLogger.info(int8Array.join(','));
int8Array[4] = -130;
AppLogger.info(int8Array.join(','));
int8Array[5] = 129;
AppLogger.info(int8Array.join(','));
  • If assigning float to int, it will just strip off everything behind decimal point 
    int8Array[6] = 1.45;
AppLogger.info(int8Array.join(','));
int8Array[7] = 1.55;
AppLogger.info(int8Array.join(','));
  • If assigning value to index which is outside of the scope of array it will be just ignored. 
    int8Array[8] = 1.55;
AppLogger.info(int8Array.join(','));

Prototype methods usage

  • copyWithin(target, start[, end])
  • It copies sequence of array elements to same array setting them on index 'target', copying starts from start, to end.
  • In this example we said, copy into index 2, starting from index 0 
    var int8ArrayNew = new Int8Array(5);
int8ArrayNew[0] = 1;
int8ArrayNew[1] = 2;
int8ArrayNew.copyWithin(2, 0);
AppLogger.info(int8Array.join(','));
  • entries()
  • Returns iterator which contains key/value pairs
    • var int8ArrayNewIterator = int8ArrayNew.entries();
for (var [key, val] of int8ArrayNewIterator) {
	AppLogger.info(key + ' : ' + val);
}
  • .every(callback(currentValue, index, array)[, thisArgument])
  • Checks each element of array against callback method. If callback returns false once, entire .every() is false, true otherwise
    • var checkForBiggerThanTwo = function(currentValue, index, array) {
	if (currentValue > 2) {
		return false;
	} else {
		retun true;
	}
};
// int8Array = [1,-123,-128,127,126,-127,1,1]
// int8ArrayNew = [1, 2, 1, 2, 0]
AppLogger.info(int8Array.every(checkForBiggerThanTwo));
AppLogger.info(int8ArrayNew.every(checkForBiggerThanTwo));
  • .fill(value[, start = 0[, end = length]])
  • Fills array with values from start (defaults to 0) to end (defaults to length)
    • var fillArray1 = new Int8Array(5);
var fillArray2 = new Int8Array(5);
var fillArray3 = new Int8Array(5);
var fillArray4 = new Int8Array(5);

fillArray1.fill(66);
fillArray2.fill(66, 1);
fillArray3.fill(66, 1, 2);
fillArray4.fill(66, 1, 55);

AppLogger.info(fillArray1.join(','));
AppLogger.info(fillArray2.join(','));
AppLogger.info(fillArray3.join(','));
AppLogger.info(fillArray4.join(','));
  • .filter(callback(currentValue, index, array)[, thisArgument])
  • Creates new typed array based on which return true in callback (which is called for each element)
    • // int8Array = [1,-123,-128,127,126,-127,1,1]
var negativeNumbersAtOddIndex = int8Array.filter(function(currentValue, index) {
	if (index % 2 === 1 && currentValue < 0) {
		return true;
	}
});
AppLogger.info(negativeNumbersAtOddIndex.join(','));
  • .find(callback(currentValue, index, array)[,thisArgument])
  • Returns first element for which callback returns true or undefined if doesn't find it.
    • // int8Array = [1,-123,-128,127,126,-127,1,1]
var isBiggerThan100 = function(currentValue, index, array) {
	if (currentValue > 100) {
		return true;
	} else {
		return false;
	}
};
AppLogger.info(int8Array.find(isBiggerThan100));
  • .findIndex(callback(currentValue, index, array)[,thisArgument])
  • Returns index of first element for which callback returns true or -1 if doesn't find it.
    • // int8Array = [1,-123,-128,127,126,-127,1,1]
var isBiggerThan100 = function(currentValue, index, array) {
	if (currentValue > 100) {
		return true;
	} else {
		return false;
	}
};
AppLogger.info(int8Array.findIndex(isBiggerThan100));
  • .forEach(callback(currentValue, index, array)[,thisArgument])
  • Executes callback for each element of typed array
    • // int8Array = [1,-123,-128,127,126,-127,1,1]
int8Array.forEach(function(currentValue, index, array) {
	if (currentValue < 0) {
		array[index]= 0;
	}
});
AppLogger.info(int8Array.join(','));
  • .indexOf(searchValue[, startingFromIndex])
  • Returns first index where it finds value given to it, or -1 if not found
  • It can start from specified index, or from beginning if not given
    • // int8Array = [1,0,0,127,126,0,1,1]
AppLogger.info(int8Array.indexOf(1, 1));
  • .join([separator])
  • Joins all elements into string, separated by separator (if given, if not it uses ',')
  • Examples shown through entire page :)
  • .keys()
  • Returns iterator which goes through all keys of array (indexes :))
    • // int8Array = [1,0,0,127,126,0,1,1]
var keys = int8Array.keys();
for (key of keys) {
	AppLogger.info(key);
}
  • .lastIndexOf(searchValue[, startingFromIndex])
  • Returns last index where it finds value given to it, or -1 if not found
  • It can start from specified index, or from the end if not given
    • // int8Array = [1,0,0,127,126,0,1,1]
AppLogger.info(int8Array.lastIndexOf(1, 3));
  • .map(callback(currentElement, index, array)[,thisArgument]
  • Creates new typed array with values modified through map function
    • // int8Array = [1,0,0,127,126,0,1,1]
var int8ArrayMinus5 = int8Array.map(function(currentElement) {
	return currentElement - 5;
});
AppLogger.info(int8ArrayMinus5.join(','));
  • .reduce(callback(previousValue, currentValue, index, array])[,initialValue])
  • Reduces all values in array by sending each of them into callback where previous value would be what we've returned in previous call (or if it's first call, value of first element (or initialValue if given)
    • // we'll sum up only last char in number
var newInt8Array = new Int8Array(4);
newInt8Array[0] = 51;
newInt8Array[1] = 51;
newInt8Array[2] = 62;
newInt8Array[3] = 72;
var reducedValue = newInt8Array.reduce(function(previousValue, currentValue, index) {
	AppLogger.info('index: ' + index);
	AppLogger.info('previousValue: ' + previousValue);
	AppLogger.info('currentValue: ' + currentValue);
	return (previousValue % 10) + (currentValue % 10);
});
AppLogger.info('reducedValue: ' + reducedValue);
  • .reduceRight(callback(previousValue, currentValue, index, array])[,initialValue])
  • It does the same as .reduce, only applies function from right-to-left
    • // we'll sum up only last char in number
newInt8Array = new Int8Array(4);
newInt8Array[0] = 51;

	AppLogger.info('previousValue: ' + previousValue);
	AppLogger.info('currentValue: ' + currentValue);
	return (previousValue % 10) + (currentValue % 10);

AppLogger.info('reducedValue: ' + reducedValue);
  • .reverse()
  • Reverses typed array in place
    • // newInt8Array = [51, 51, 62, 72]
newInt8Array.reverse();
AppLogger.info(newInt8Array.join(','));
  • .set(array[,offset])
  • Stores values in array based on array (or typed array given)
  • Offset determines where to start storing (length of given array + offset must be under length of created array)
    • var setArrayExample = new Int8Array(4);
setArrayExample.set([1,2,3], 1);
AppLogger.info(setArrayExample.join(','));
  • .slice(begin[,end])
  • Extracts element from typed array and creates a new one
    • // setArrayExample = [0,1,2,3]
var slicedArray = setArrayExample.slice(1,3);
AppLogger.info(slicedArray.join(','));
  • .some(callback(currentElement, index, array)[,thisArg])
  • Returns true if one of the elements in callback returns true
    • // setArrayExample = [0,1,2,3]
var isThereATwo = function(currentElement) {
	if (currentElement === 2) {
		return true;
	} else {
		return false;
	}
};
AppLogger.info(setArrayExample.some(isThereATwo));
  • .sort([sortingFunction(a,b)]) - NOT SUPPORTED CURRENTLY
  • Sorts array in place.
  • It can have sortingFunction which receives two elements and implements sorting algorithm
  • In following example does simple ascending sorting and later descending sorting 
    // newInt8Array = [72, 62, 51, 51];
var ascSort = function(a, b) {
	if (a > b) {
		return 1;
	} else if (a === b) {
		return 0;
	} else {
		return -1;
	}
};
var descSort = function(a, b) {
	if (a > b) {
		return -1;
	} else if (a === b) {
		return 0;
	} else {
		return 1;
	}
};
newInt8Array.sort(ascSort);
// newInt8Array = [51, 51, 62, 72];
AppLogger.info(newInt8Array.join(','));
newInt8Array.sort(descSort);
// newInt8Array = [72, 62, 51, 51];
AppLogger.info(newInt8Array.join(','));
  • .subarray([begin[,end]])
  • Returns new array based on the same ArrayBuffer. Similar to splice, but it doesn't change array in place
  • end index not included
    • // newInt8Array = [72, 62, 51, 51];	
var subarray = newInt8Array.subarray(1,3);
AppLogger.info(subarray.join(','));
  • .values()
  • Returns iterator which holds values of type array
    • // newInt8Array = [72, 62, 51, 51];
var values = newInt8Array.values();
for (val of values) {
	AppLogger.info(val);
}
  • .toLocaleString()
  • Returns local string representation of elements inside of the array
    • // newInt8Array = [72, 62, 51, 51];
AppLogger.info(newInt8Array.toLocaleString());
  • .toString()
  • Returns string representation of elements inside of the array
    • // newInt8Array = [72, 62, 51, 51];
