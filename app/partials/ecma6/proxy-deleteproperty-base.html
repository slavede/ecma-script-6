• Trap for delete operator which deletes property from the object (delete object[property])
  • It gets two parameters
    • target - target object
    • property - property we are trying to delete
  • Return true/false depending if deleting was successful or not (in strict mode, false will thorw TypeError exception)
var myObject = {
	name : 'MyName',
	height : 180,
};

var proxy = new Proxy(myObject, {
	deleteProperty : function(target, prop) {
		AppLogger.info('Deleting property ' + prop, 'proxy-deleteproperty-1');
		if (prop === 'name') {
			return false;
		}
		delete target[prop];
		return true;
	}
});

AppLogger.info(delete proxy.name);
AppLogger.info('myObject.name: ' + myObject.name, 'proxy-deleteproperty-2');

AppLogger.info(delete proxy.height);
AppLogger.info('myObject.height: ' + myObject.height, 'proxy-deleteproperty-3');
