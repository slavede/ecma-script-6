function myTagger(strings, value1, value2) {
	return 'raw: ' + strings.raw.join(',') + '\n strings: ' + strings.join(',') + '\n values: ' + value1 + ' - ' + value2 ;
}

AppLogger.info(myTagger`User ${name} will use\n my tagger here for ${a}st time`);
